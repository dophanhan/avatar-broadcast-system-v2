    const DB_VERSION = 1;
    const STORE_NAME = 'videos';
    // 打开数据库
    function openDB() {
      return new Promise((resolve, reject) => {
        const request = indexedDB.open(DB_NAME, DB_VERSION);
        request.onupgradeneeded = (event) => {
          const db = event.target.result;
          if (!db.objectStoreNames.contains(STORE_NAME)) {
            db.createObjectStore(STORE_NAME, { keyPath: 'id', autoIncrement: true });
          }
        };
        request.onsuccess = () => resolve(request.result);
        request.onerror = () => reject(request.error);
      });
    }
    // 保存视频到缓存
    async function saveVideoToCache(videoBlob, text, duration) {
      const db = await openDB();
      return new Promise((resolve, reject) => {
        const transaction = db.transaction([STORE_NAME], 'readwrite');
        const store = transaction.objectStore(STORE_NAME);
        const record = {
          video: videoBlob,
          text: text,
          duration: duration,
          timestamp: Date.now()
        };
        const request = store.add(record);
        request.onsuccess = () => resolve(request.result);
        request.onerror = () => reject(request.error);
      });
    }
    // 获取所有缓存视频
    async function getAllCachedVideos() {
      const db = await openDB();
      return new Promise((resolve, reject) => {
        const transaction = db.transaction([STORE_NAME], 'readonly');
        const store = transaction.objectStore(STORE_NAME);
        const request = store.getAll();
        request.onsuccess = () => resolve(request.result || []);
        request.onerror = () => reject(request.error);
      });
    }
    // 删除缓存视频
    async function deleteCachedVideo(id) {
      const db = await openDB();
      return new Promise((resolve, reject) => {
        const transaction = db.transaction([STORE_NAME], 'readwrite');
        const store = transaction.objectStore(STORE_NAME);
        const request = store.delete(id);
        request.onsuccess = () => resolve();
        request.onerror = () => reject(request.error);
      });
    }

        function blobToBase64(blob) {
          return new Promise((resolve, reject) => {
            const reader = new FileReader();
            reader.onloadend = () => resolve(reader.result);
            reader.onerror = reject;
            reader.readAsDataURL(blob);
          });

        // 【核心重写】彻底放弃 requestAnimationFrame,改用 setInterval 强制 25fps 绘制(无视后台降频)
        function compositeFrame() {
          // 如果已经有定时器在跑,先清掉
          if (compositeInterval) {
            clearInterval(compositeInterval);
            compositeInterval = null;
          }

          addLog('🎨 启动强制绘制模式 (25fps 固定帧率,无视后台降频)...', 'info');

          // 强制 25fps,即每 40ms 绘制一帧
          const FRAME_INTERVAL = 40;

          compositeInterval = setInterval(() => {
            // 如果不在录制或播报状态,立刻停止定时器
            if (!isRecording.value && !isBroadcasting.value) {
              clearInterval(compositeInterval);
              compositeInterval = null;
              addLog('⏹️ 强制绘制模式已停止', 'info');
              return;
            }

            if (!compositeCanvas.value) return;
            const canvas = compositeCanvas.value;
            const ctx = canvas.getContext('2d');
            const videoEl = document.querySelector('#digital-human-container video');

            try {
              // 1. 强制清空画布(防止帧残留)
              ctx.clearRect(0, 0, canvas.width, canvas.height);


              // ========== 新增:PPT 底层绘制 ==========
              if (broadcastMode.value === 'horizontal' && pptConfig.imported && pageTurningConfirmed.value) {
                const currentPageData = pptConfig.pages.find(item => item.page_num === currentPptPage.value);
                if (currentPageData && currentPageData.page_image) {
                  ctx.drawImage(currentPageData.page_image, 0, 0, 1344, 1080);
                } else {
                  ctx.fillStyle = '#1a1a1a';
                  ctx.fillRect(0, 0, 1344, 1080);
                }
              }
              // ========== 新增结束 ==========
              // 2. 如果视频未就绪,画黑屏等待
              if (!videoEl || videoEl.readyState < 2) {
                ctx.fillStyle = '#000';
                ctx.fillRect(0, 0, canvas.width, canvas.height);
                ctx.fillStyle = '#fff';
                ctx.font = '24px Microsoft Yahei';
                ctx.textAlign = 'center';
                ctx.fillText('等待数字人...', canvas.width/2, canvas.height/2);
                return;
              }

              // 3. 强制绘制当前视频帧(不管有没有新帧,都画一次,保证时间戳连续)
              if (broadcastMode.value === 'horizontal') {
                ctx.drawImage(videoEl, 1344, 0, 576, 1080);
              } else {
                ctx.drawImage(videoEl, 0, 0, canvas.width, canvas.height);
              }

              // 4. 绘制字幕
              if (currentSubtitle) {
                ctx.font = 'bold 28px Microsoft Yahei';
                ctx.fillStyle = '#ffffff';
                ctx.textAlign = 'center';
                ctx.textBaseline = 'bottom';
                ctx.shadowColor = '#000000';
                ctx.shadowBlur = 6;
                ctx.shadowOffsetX = 2;
                ctx.shadowOffsetY = 2;

                const maxWidth = canvas.width * 0.9;
                const metrics = ctx.measureText(currentSubtitle);

                if (metrics.width <= maxWidth) {
                  const subtitleY = canvas.height - 50;
                  ctx.fillStyle = 'rgba(0, 0, 0, 0.5)';
                  ctx.fillRect(
                    (canvas.width - metrics.width) / 2 - 20,
                    subtitleY - 35,
                    metrics.width + 40,
                    40
                  );
                  ctx.fillStyle = '#ffffff';
                  ctx.fillText(currentSubtitle, canvas.width / 2, subtitleY);
                } else {
                  const lines = [];
                  let line = '';
                  const words = currentSubtitle.split('');
                  words.forEach(word => {
                    const testLine = line + word;
                    const m = ctx.measureText(testLine);
                    if (m.width > maxWidth && line) {
                      lines.push(line);
                      line = word;
                    } else {
                      line = testLine;
                    }
                  });
                  lines.push(line);
                  const lineHeight = 36;
                  const subtitleBottom = canvas.height - 50;
                  const bgHeight = lines.length * lineHeight + 20;
                  const bgY = subtitleBottom - bgHeight + 10;
                  ctx.fillStyle = 'rgba(0, 0, 0, 0.5)';
                  ctx.fillRect(canvas.width * 0.05, bgY, canvas.width * 0.9, bgHeight);
                  ctx.fillStyle = '#ffffff';
                  lines.forEach((line, index) => {
                    const y = subtitleBottom - (lines.length - index - 1) * lineHeight;
                    ctx.fillText(line, canvas.width / 2, y);
                  });
                }
                ctx.shadowColor = 'transparent';
                ctx.shadowBlur = 0;
                ctx.shadowOffsetX = 0;
                ctx.shadowOffsetY = 0;
              }
            } catch (e) {
              // 忽略绘制异常,防止单帧错误导致整个录制崩溃
              // addLog('绘制异常:' + e.message, 'warn');
            }
          }, FRAME_INTERVAL);
        }


        // 录制相关函数
        function startRecording() {
          if (isRecording.value) {
            addLog('已经在录制中', 'warn');
            return;
          }

          // ============== 【新增】长视频内存监控预警 ==============
          if (performance && performance.memory) {
            const memoryCheckTimer = setInterval(() => {
              if (!isRecording.value) {
                clearInterval(memoryCheckTimer); // 停止录制时自动结束监控
                return;
              }
              const usedMB = (performance.memory.usedJSHeapSize / 1024 / 1024).toFixed(0);
              const limitMB = (performance.memory.jsHeapSizeLimit / 1024 / 1024).toFixed(0);

              // 超过 1.5GB 发出橙色预警,超过 2GB 发出红色预警
              if (usedMB > 2000) {
                addLog(`🔴 内存严重不足:${usedMB}MB / ${limitMB}MB,建议立即停止录制并分段`, 'error');
              } else if (usedMB > 1500) {
                addLog(`🟠 内存使用过高:${usedMB}MB / ${limitMB}MB`, 'warn');
              }
            }, 5000); // 每 5 秒检查一次
          }
          // ============== 【新增】结束 ==============

          addLog('🎬 开始初始化录制...', 'info');
          recordedChunks = [];
          isRecording.value = true;
          recordStartTime = Date.now();
          recordDuration.value = 0;

          recordTimer = setInterval(() => {
            recordDuration.value = Math.floor((Date.now() - recordStartTime) / 1000);
          }, 1000);

          // 【关键修复】先启动画面合成循环(确保 Canvas 有内容)
          addLog('🎨 启动 Canvas 合成循环(60fps)...', 'info');
          compositeFrame();

          // 延迟 200ms 启动录制(给 Canvas 一点时间渲染第一帧)
          setTimeout(() => {
            setupMediaRecorderNow(() => {
              addLog('✅ 录制器就绪,可以开始播报', 'success');
            });
          }, 200);
        }

                // 注意:这里增加了一个 onRecorderReady 回调参数
        // 【核心修复】分离式捕获:Canvas(视频 + 字幕) + Audio(纯音频) → 合并录制
        function setupMediaRecorderNow(onRecorderReady) {
          addLog('📹 启动 MediaRecorder(字幕 + 视频 + 音频三合一版)...', 'info');

          const canvas = compositeCanvas.value;
          if (!canvas) {
            addLog('❌ Canvas 不存在', 'error');
            return;
          }

          // 1. 获取 Canvas 视频流(60fps 高帧率,确保流畅)
          const canvasStream = canvas.captureStream(25);
          addLog(`🎬 Canvas 视频流已捕获:${canvasStream.getVideoTracks().length} 个视频轨`, 'success');

          // 2. 定义启动录制的核心函数
          const startRecordingWhenReady = () => {
            let audioStream = null;
            const videoEl = document.querySelector('#digital-human-container video');
            const allAudios = document.querySelectorAll('audio');

            // 优先从 video 元素获取音频(讯飞 SDK 通常用 video 元素)
            if (videoEl && videoEl.readyState >= 3 && !videoEl.muted) {
              try {
                const tempStream = videoEl.captureStream ? videoEl.captureStream() : (videoEl.mozCaptureStream ? videoEl.mozCaptureStream() : null);
                if (tempStream && tempStream.getAudioTracks().length > 0) {
                  audioStream = new MediaStream(tempStream.getAudioTracks());
                  addLog(`✅ 从 video 元素获取到音频流 (${audioStream.getAudioTracks().length} 轨)`, 'success');
                }
              } catch(e) {
                addLog(`⚠️ 从 video 获取音频失败:${e.message}`, 'warn');
              }
            }

            // 如果 video 没有音频,再尝试找 audio 元素
            if (!audioStream || audioStream.getAudioTracks().length === 0) {
              for (let i = 0; i < allAudios.length; i++) {
                const a = allAudios[i];
                if (!a.muted) {
                  try {
                    const tempStream = a.captureStream ? a.captureStream() : (a.mozCaptureStream ? a.mozCaptureStream() : null);
                    if (tempStream && tempStream.getAudioTracks().length > 0) {
                      audioStream = new MediaStream(tempStream.getAudioTracks());
                      addLog(`✅ 从第 ${i+1} 个 audio 元素获取到音频流`, 'success');
                      break;
                    }
                  } catch(e) {}
                }
              }
            }

            // 如果找到了音频,立刻开始录制;如果没找到,过 200ms 再试一次
            if (audioStream && audioStream.getAudioTracks().length > 0) {
              createRecorderAndStart(audioStream);
            } else {
              addLog('⏳ 等待音频轨道就绪...', 'info');
              setTimeout(startRecordingWhenReady, 200);
            }
          };

          // 创建录制器并启动
          const createRecorderAndStart = (audioStream) => {
            // 合并 Canvas 视频轨 + 音频轨
            const combinedStream = new MediaStream([
              ...canvasStream.getVideoTracks(),
              ...audioStream.getAudioTracks()
            ]);

            const videoTracks = combinedStream.getVideoTracks();
            const audioTracks = combinedStream.getAudioTracks();
            addLog(`🎞️ 最终流状态:视频轨 ${videoTracks.length} 个,音频轨 ${audioTracks.length} 个`, 'success');

            // 检查视频轨和音频轨是否都存在
            if (videoTracks.length === 0) {
              addLog('❌ 没有视频轨道!', 'error');
              return;
            }
            if (audioTracks.length === 0) {
              addLog('⚠️ 没有音频轨道,将录制无声视频', 'warn');
            }

            // 选择最佳编码格式(优先尝试 H.264 编码,兼容性最好)
            let options = null;

            // 【方案 1】优先尝试 H.264 编码 (webm 容器封装,兼容性最接近 MP4)
            const h264Type = 'video/webm;codecs=h264,opus';
            if (MediaRecorder.isTypeSupported(h264Type)) {
              options = {
                mimeType: h264Type,
                videoBitsPerSecond: 1000000, // H.264 稍微给高点码率,画质更好
                audioBitsPerSecond: 128000    // H.264 配 128kbps AAC 音质
              };
              addLog('✅ 使用 H.264 编码 (最佳兼容性)', 'success');
            }
            // 【方案 2】如果不支持 H.264,退回到最稳的 VP8
            else {
              options = {
                mimeType: 'video/webm;codecs=vp8,opus',
                videoBitsPerSecond: 800000,
                audioBitsPerSecond: 64000
              };
              addLog('⚠️ 浏览器不支持 H.264,使用 VP8 编码', 'warn');
            }

            addLog(`📦 使用编码格式:${options.mimeType}`, 'info');

            mediaRecorder = new MediaRecorder(combinedStream, options);

            mediaRecorder.ondataavailable = (e) => {
              if (e.data.size > 0) {
                recordedChunks.push(e.data);

                // 【优化 1】日志瘦身:只保留最后 50 条日志,防止日志占用内存
                if (logs.value.length > 200) {
                  logs.value = logs.value.slice(-50);
                }

                // 【优化 2】实时显示内存累计占用,心里有底
                const totalMemMB = recordedChunks.reduce((sum, chunk) => sum + chunk.size, 0) / 1024 / 1024;
                addLog(`📥 录制数据块:${(e.data.size / 1024).toFixed(1)} KB (内存累计:${totalMemMB.toFixed(1)} MB)`, 'info');
              }
            };

            mediaRecorder.onstop = () => {
              addLog(`🛑 录制停止,数据块数量:${recordedChunks.length}`, 'success');
              if (animationFrame) {
                cancelAnimationFrame(animationFrame);
                addLog('⏹️ Canvas 合成循环已停止', 'info');
              }

              videoBlob.value = new Blob(recordedChunks, { type: mediaRecorder.mimeType });

              if (videoBlob.value.size > 0) {
                const fileSize = (videoBlob.value.size / 1024 / 1024).toFixed(2);
                addLog(`💾 视频生成成功:${fileSize} MB`, 'success');

                const videoUrl = URL.createObjectURL(videoBlob.value);
                previewVideo.value.src = videoUrl;

                // 验证视频内容
                previewVideo.value.onloadeddata = () => {
                  addLog(`✅ 视频已加载:时长 ${formatDuration(previewVideo.value.duration)},尺寸 ${previewVideo.value.videoWidth}x${previewVideo.value.videoHeight}`, 'success');

                  // 检查是否有音频
                  if (previewVideo.value.mozHasAudio || previewVideo.value.webkitAudioDecodedByteCount > 0) {
                    addLog('🔊 视频包含音频轨道', 'success');
                  } else {
                    addLog('⚠️ 视频可能没有音频', 'warn');
                  }
                };

                previewVideo.value.onerror = (e) => {
                  addLog(`❌ 视频加载失败:${JSON.stringify(e)}`, 'error');
                };
              } else {
                addLog('❌ 视频 Blob 为空!', 'error');
              }
            };

            mediaRecorder.start(100);
            addLog('🔴 MediaRecorder 已启动(25fps 视频 + 音频)', 'success');

            // 回调通知可以开始播报
            if (onRecorderReady) {
              onRecorderReady();
            }
          };

          // 立刻开始第一轮检测
          startRecordingWhenReady();
        }

        function setupMediaRecorder(videoStream, audioStream) {
          const combinedStream = new MediaStream([
            ...videoStream.getVideoTracks(),
            ...audioStream.getAudioTracks()
          ]);

          try {
            mediaRecorder = new MediaRecorder(combinedStream, {
              mimeType: 'video/webm;codecs=vp9',
              videoBitsPerSecond: 5000000
            });
          } catch (e) {
            mediaRecorder = new MediaRecorder(combinedStream, {
              mimeType: 'video/webm',
              videoBitsPerSecond: 5000000
            });
          }

          mediaRecorder.ondataavailable = (e) => {
            if (e.data.size > 0) {
              recordedChunks.push(e.data);
              addLog(`录制数据块:${e.data.size} bytes`, 'info');
            }
          };

          mediaRecorder.onstop = () => {
            addLog(`录制停止,数据块数量:${recordedChunks.length}`, 'success');
            if (animationFrame) cancelAnimationFrame(animationFrame);

            const mimeType = mediaRecorder.mimeType;
            videoBlob.value = new Blob(recordedChunks, { type: mimeType });

            if (videoBlob.value.size > 0) {
              const fileSize = (videoBlob.value.size / 1024 / 1024).toFixed(2);
              addLog(`视频生成成功:${fileSize} MB`, 'success');

              const videoUrl = URL.createObjectURL(videoBlob.value);
              previewVideo.value.src = videoUrl;

              // 等待 video 元素加载完成
              previewVideo.value.onloadeddata = () => {
                addLog(`✅ 视频已加载完成,时长:${formatDuration(previewVideo.value.duration)}`, 'success');
              };

              previewVideo.value.onerror = () => {
                addLog('❌ 视频加载失败', 'error');
              };
            } else {
              addLog('视频 Blob 为空!', 'error');
            }
          };

          mediaRecorder.start(100);
          addLog('MediaRecorder 已启动', 'success');
        }

        function stopRecording() {
          if (!isRecording.value) {
            addLog('没有在录制', 'warn');
            return;
          }

          addLog('准备停止录制...', 'info');

          // 停止 mediaRecorder(触发 onstop,生成 videoBlob)
          if (mediaRecorder && mediaRecorder.state !== 'inactive') {
            mediaRecorder.stop();
            addLog('🛑 MediaRecorder 已停止', 'success');
          } else if (!mediaRecorder) {
            addLog('⚠️ mediaRecorder 不存在,直接清理状态', 'warn');
          }

          isRecording.value = false;

          if (recordTimer) {
            clearInterval(recordTimer);
            recordTimer = null;
          }

          if (animationFrame) {
            cancelAnimationFrame(animationFrame);
            animationFrame = null;
          }

          // 【新增】强制清理绘制定时器
          if (compositeInterval) {
            clearInterval(compositeInterval);
            compositeInterval = null;
            addLog('⏹️ 强制绘制定时器已清理', 'info');
          }

          addLog('⏹️ Canvas 合成已停止', 'info');
        }

        // 播报结束处理(自然完成)

          if (!isBroadcasting.value && !isRecording.value) return;

          addLog('🏁 检测到播报完全结束,停止录制', 'success');
          stopRecording();

          isBroadcasting.value = false;
          isBroadcastingSentence = false;
          currentSentence.value = 0;
          totalSentences.value = 0;
          progressPercent.value = 0;
          waitingForBroadcastEnd = false;
          sentenceIndex = 0;
          currentSentences = [];
          currentSubtitle = '';

          // 等待 videoBlob 生成
          setTimeout(() => {
            if (videoBlob.value && videoBlob.value.size > 0) {
              addLog(`🎉 播报流程全部完成!视频已准备好下载`, 'success');
            } else {
              addLog('⚠️ 播报完成,但视频可能未生成', 'warn');
            }
          }, 500);
        }

        // 播报失败处理(异常中断)
        function onBroadcastFailed(reason) {
          if (isStopping) return;  // 正在停止中,不处理
          addLog(`❌ 播报失败:${reason}`, 'error');

          if (isRecording.value) {
            addLog('💾 保存已录制的视频...', 'warn');
            stopRecording();  // 这会触发 mediaRecorder.onstop,生成 videoBlob
          }

          isBroadcasting.value = false;
          isBroadcastingSentence = false;
          currentSentence.value = 0;
          totalSentences.value = 0;
          progressPercent.value = 0;

          // 等待 videoBlob 生成
          setTimeout(() => {
            if (videoBlob.value && videoBlob.value.size > 0) {
              addLog(`✅ 已录制的视频已保存,可以下载`, 'success');
            } else {
              addLog('⚠️ 视频可能未生成', 'warn');
            }
          }, 500);
        }

        // 播报单句
        async function speakSingleSentence(text) {
          if (!sdkInstance) {
            throw new Error('SDK 未初始化');
          }

          // 等待队列空闲(避免发送太快导致队列满)
          while (isBroadcastingSentence) {
            await new Promise(resolve => setTimeout(resolve, 200));
          }

          isBroadcastingSentence = true;
          currentSubtitle = text.trim();
          addLog(`🎤 正在播报句子:${text}`, 'info');

          try {
            const requestId = await sdkInstance.writeText(text, {
              nlp: false,
              avatar_dispatch: { interactive_mode: 0 },
              tts: {
                vcn: config.rdn,
                speed: Math.floor(config.speed * 50),
                volume: 100,
                pitch: 50
              }
            });
            addLog(`✅ writeText 成功,requestId: ${requestId}`, 'success');

            // 等待播报完成(frame_stop 事件会设置 isBroadcastingSentence = false)
            while (isBroadcastingSentence) {
              await new Promise(resolve => setTimeout(resolve, 100));
            }

            addLog(`🏁 句子播报完成`, 'success');
          } catch (err) {
            isBroadcastingSentence = false;
            addLog(`❌ writeText 失败:${JSON.stringify(err)}`, 'error');
            // 如果是连接错误,触发失败处理,保存已录制的视频
            if (err.code === 'InvalidConnect' || err.code === '600003') {
              onBroadcastFailed('连接异常,已保存已录制的视频');
            }
            throw err;
          }
        }

        // 等待视频元素就绪
        async function waitForVideoElement() {
          return new Promise((resolve) => {
            const maxWaitTime = 15000;
            const startTime = Date.now();

            const check = () => {
              if (Date.now() - startTime > maxWaitTime) {
                addLog('等待 video 元素超时', 'warn');
                resolve();
                return;
              }
              const videoEl = document.querySelector('#digital-human-container video');
              if (videoEl && videoEl.readyState >= 2 && videoEl.videoWidth > 0) {
                addLog(`Video 元素就绪 (${videoEl.videoWidth}x${videoEl.videoHeight})`, 'success');
                resolve();
              } else {
                setTimeout(check, 500);
              }
            };
            check();
          });
        }

        // 初始化 SDK
        async function initSDK(authData) {
          if (sdkInstance) {
            sdkInstance.stop();
            sdkInstance.destroy();
            sdkInstance = null;
          }

          sdkInstance = new window.AVATAR_SDK.AvatarSDK({
            logLevel: 1,
            useInlinePlayer: true
          });

          // 事件监听
          sdkInstance.on('connected', () => {
            addLog('数字人连接成功', 'success');
            isConnected.value = true;
          });

          sdkInstance.on('error', (err) => {
            // 敏感词错误处理
            if (err.code === 20017 || (err.message && err.message.includes('illegal text'))) {
              addLog('❌ 文本包含敏感词汇,请修改后重试', 'error');
              addLog('提示:避免政治、色情、暴力等敏感内容', 'warn');
            } else if (err.code === 10110 && err.message && err.message.includes('word contain sensitive')) {
              addLog('❌ 文本包含敏感词汇,请修改后重试', 'error');
            } else {
              addLog('数字人连接失败:' + JSON.stringify(err), 'error');
            }
            stopBroadcast();
          });

          sdkInstance.on('disconnected', () => {
            addLog('数字人连接断开', 'warn');
            isConnected.value = false;
          });

          sdkInstance.on('render-start', () => {
            addLog('渲染开始', 'success');
          });

          sdkInstance.on('frame_start', (data) => {
            lastFrameStartTime = Date.now();

            // 更新当前播报的句子(用于字幕显示)
            if (sentenceIndex < currentSentences.length) {
              currentSubtitle = currentSentences[sentenceIndex];
              addLog(`📝 字幕更新 [句${sentenceIndex + 1}/${currentSentences.length}]: ${currentSubtitle.substring(0, 30)}...`, 'info');
              sentenceIndex++;
            }

            if (waitingForBroadcastEnd && broadcastEndTimer) {
              clearTimeout(broadcastEndTimer);
              broadcastEndTimer = null;
              waitingForBroadcastEnd = false;
            }
          });

                    sdkInstance.on('frame_stop', (data) => {
            if (data?.vmr_status === 2) {
              addLog('🏁 单句播报完成', 'success');
              isBroadcastingSentence = false; // 释放队列锁,让下一句可以进来

              // 【核心修复】只有当真正的最后一句也播报完了,才停止录制
              // 注意:这里的判断必须非常严格:
              // 1. sentenceIndex 必须大于等于句子总数(说明都读完了)
              // 2. 并且我们确实还在播报状态中
              if (sentenceIndex >= currentSentences.length && isBroadcasting.value) {
                addLog(`✅ 所有句子已播报完成 (${sentenceIndex}/${currentSentences.length}),准备结束录制`, 'success');
                // 稍微延迟一下,确保最后一帧画面录进去
                setTimeout(() => {
                  onBroadcastEnded();
                }, 500);
              }
            }
          });

          sdkInstance.setApiInfo({
            appId: authData.appId,
            sceneId: authData.sceneId,
            signedUrl: authData.signedUrl
          });

          // ✅ 修复:关闭透明通道,解决警告
          const isSuperRealistic = config.roleId.startsWith('cnr');
          sdkInstance.setGlobalParams({
            stream: {
              protocol: 'xrtc',
              fps: 25,
              bitrate: 1000000,
              alpha: 0
            },
            avatar: {
              avatar_id: authData.avatarId || config.roleId,
              width: 1080,
              height: 1920,
              audio_format: 1
            },
            tts: {
              vcn: authData.vcn || config.rdn,
              speed: Math.floor(config.speed * 50),
              pitch: 50,
              volume: 100
            },
            avatar_dispatch: {
              interactive_mode: 1
            },
            air: {
              air: 1,
              add_nonsemantic: 1
            }
          });

          // 启动 SDK
          await sdkInstance.start({
            wrapper: document.getElementById('digital-human-container')
          });

          await waitForVideoElement();

          // 确保 video 元素不静音,音量最大(关键修复:确保音频可捕获)
          const videoEl = document.querySelector('#digital-human-container video');
          if (videoEl) {
            videoEl.muted = false;
            videoEl.volume = 1.0;
            videoEl.autoplay = true;
            videoEl.playsInline = true;

            // 禁用所有可能影响音频的属性
            videoEl.removeAttribute('muted');

            addLog('🔊 Video 元素已配置:muted=false, volume=1.0, autoplay=true', 'success');

            // 监听视频元素事件
            videoEl.onplay = () => addLog('▶️ Video 开始播放', 'success');
            videoEl.onpause = () => addLog('⏸️ Video 暂停', 'warn');
            videoEl.onvolumechange = () => addLog(`🔊 Volume 变化:${videoEl.volume}, muted=${videoEl.muted}`, 'info');
          } else {
            addLog('⚠️ Video 元素未找到', 'warn');
          }

          addLog('SDK 启动成功', 'success');
        }

        // 开始播报
        async function startBroadcast() {
          if (!config.text) {
            addLog('请输入播报文本', 'warn');
            return;
          }

          // 敏感词提醒(逐句播报,每句不会超过 2000 字,无需检查总长度)
          const sensitiveKeywords = ['政治', '色情', '暴力', '赌博', '毒品'];
          const foundKeywords = sensitiveKeywords.filter(keyword => config.text.includes(keyword));
          if (foundKeywords.length > 0) {
            addLog(`⚠️ 文本可能包含敏感内容:${foundKeywords.join(', ')}`, 'warn');
            addLog('如播报失败,请修改文本后重试', 'info');
          }

          if (isBroadcasting.value) return;

          isBroadcasting.value = true;
          currentSentence.value = 0;
          totalSentences.value = 0;
          progressPercent.value = 0;
          videoBlob.value = null;
          currentSubtitle = '';
          waitingForBroadcastEnd = false;
          broadcastEndTimer = null;

          addLog('🎬 开始播报流程', 'success');

          try {
            // 获取鉴权信息
            addLog('正在获取鉴权信息...', 'info');
            const authResponse = await fetch('https://avatar-roadcast-shakmjgzdi.cn-hangzhou.fcapp.run', {
              method: 'POST',
              headers: { 'Content-Type': 'application/json' },
              body: JSON.stringify({
                avatarId: config.roleId,
                vcn: config.rdn,
                fullText: config.text
              })
            });

            if (!authResponse.ok) {
              throw new Error('获取鉴权信息失败:' + authResponse.statusText);
            }
            const authData = await authResponse.json();
            if (authData.code !== 0) {
              throw new Error(authData.msg || '鉴权失败');
            }
            addLog('鉴权信息获取成功', 'success');

            // 初始化 SDK
            await initSDK(authData.data);

            // 拆分句子 - 使用 Unicode 代码点匹配中文标点(保留标点符号)
            const fullText = config.text;
            // \u3002=。, \uFF01=!, \uFF1F=?, \u002E=., \u0021=!, \u003F=?
            // 使用 replace 在标点后添加特殊标记,然后拆分
            const sentences = fullText
              .replace(/([\u3002\uFF01\uFF1F\u002E\u0021\u003F]+)/g, '$1|SPLIT|')
              .split('|SPLIT|')
              .map(s => s.trim())
              .filter(s => s.length > 0);

            totalSentences.value = sentences.length;
            addLog(`📝 逐句拆分完成,共 ${sentences.length} 句`, 'success');
            sentences.forEach((s, i) => {
              addLog(`  句${i + 1}: ${s.substring(0, 50)}${s.length > 50 ? '...' : ''}`, 'info');
            });

            // 启动录制
            startRecording();

            // 设置当前句子列表(用于 frame_start 事件更新字幕)
            currentSentences = sentences;
            sentenceIndex = 0;
            currentSubtitle = '';

            // 逐句播报
            for (let i = 0; i < sentences.length; i++) {
              // 检查是否停止
              if (isStopping) {
                addLog('⚠️ 用户停止播报,中断循环', 'warn');
                break;
              }

              currentSentence.value = i + 1;
              progressPercent.value = Math.round(((i + 1) / sentences.length) * 100);
              await speakSingleSentence(sentences[i]);
            }

            addLog('📡 所有句子已发送,等待播报结束', 'info');
          } catch (e) {
            addLog('播报失败:' + e.message, 'error');
            onBroadcastFailed(e.message);
          }
        }

        // 停止播报
        async function stopBroadcast() {
          if (isStopping) {
            addLog('⚠️ 正在停止中,请勿重复操作', 'warn');
            return;
          }

          isStopping = true;
          addLog('用户手动停止播报', 'warn');

          if (broadcastEndTimer) {
            clearTimeout(broadcastEndTimer);
            broadcastEndTimer = null;
          }

          if (sdkInstance) {
            try {
              sdkInstance.interrupt();
              sdkInstance.stop();
              // 不调用 destroy,保留 video 元素用于预览
              // sdkInstance.destroy();
            } catch (e) {
              addLog('SDK 停止异常:' + e.message, 'warn');
            }
            // sdkInstance = null;  // 保留实例
          }

          stopRecording();
          isBroadcasting.value = false;
          isConnected.value = false;
          isBroadcastingSentence = false;
          currentSentence.value = 0;
          totalSentences.value = 0;
          progressPercent.value = 0;
          currentSubtitle = '';

          // 重置停止标志
          setTimeout(() => {
            isStopping = false;
          }, 1000);
        }

        // 生命周期
        onMounted(async () => {

        // 全屏切换方法

          }
        }

        };
      }
